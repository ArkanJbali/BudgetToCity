  getRndInteger(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
  }

  getPlusOrMinus() {
    return Math.random() < 0.5 ? -1 : 1;

    var to = this.data['toDestination']['airportCode'] + "-sky";
